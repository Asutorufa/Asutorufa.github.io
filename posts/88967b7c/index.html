利用OpenCV自带的分类器识别面部及眼睛

OpenCV自带里许多分类器,可在haarcascades下载

获取面部

1// 获取面部2// image 为输入的OpenCV的格式的图像3// out 为输出图像 <- 此处输出图像为裁剪下来的矩形面部图像4bool get_face(cv::Mat image, cv::Mat &out)5{6    // 将图像转为灰度图像,可提高识别精度7    cv::Mat gray;8    cv::cvtColor(image, gray, CV_BGR2GRAY, 0);9    // 此处的rect为用来存储识别出的面部的定位 <- 这里为vector是因为识别的面部可能不止为一个10    std::vector<cv::Rect> rect;11    // 初始化OpenCV的分类器12    cv::CascadeClassifier cas = cv::CascadeClassifier("haarcascade_frontalface_default.xml");13    cas.detectMultiScale(gray, rect, 1.15, 5);14    // 判断rect是否为空,为空的话就是没有被识别的面部15    if (rect.empty())16    {17        return false;18    }19    // 输出图像,这里的rect可能不止为一个(因为可能不止一张脸),所以实际中应该使用for循环20    // 我这里用rect[0]为了简化21    out = image(rect[0]);22    return true;23}

获取眼睛

1// 这里与上面相同(这里我只识别左眼,OpenCV还提供了其他文件可获取右眼和双眼)2bool get_left_eye(cv::Mat image, cv::Mat &out)3{4    // 之后与面部相同5    cv::Mat gray;6    cv::cvtColor(image, gray, CV_BGR2GRAY, 0);7    std::vector<cv::Rect> rect;8    cv::CascadeClassifier cas = cv::CascadeClassifier("xml/haarcascade_lefteye_2splits.xml");9    cas.detectMultiScale(gray, rect, 1.15, 5);10    if (rect.empty())11    {12        return false;13    }14    out = image(rect[0]);15    return true;16}

瞳孔定位

  • 获取的眼睛图像
  • 裁剪去除眉毛
  • 伽马矫正提高图像对比度
  • 转化为灰度图像
  • 利用threshold进行阈值操作(此处可能会改为inRange,目前还在测试)
  • 腐蚀
  • 膨胀
  • medianBlur平滑图像
  • 检测关键点并画在图像上
1// 定位眼球 -> 检测关键点2std::vector<cv::KeyPoint> dector(cv::Mat image)3{4    cv::SimpleBlobDetector::Params detector_params;5    detector_params.filterByArea = true;6    detector_params.maxArea = 1500;7    cv::Ptr<cv::SimpleBlobDetector> blob_dector = cv::SimpleBlobDetector::create(detector_params);8    std::vector<cv::KeyPoint> keypoints;9    blob_dector->detect(image, keypoints);10    return keypoints;11}12 13bool get_pupil(cv::Mat image, cv::Mat &out)14{15    // width -> image.rows ;16    // height -> image.cols;17    // cut eyebrow 裁剪 去除眉毛18    image = image(cv::Rect(image.cols / 10, image.rows / 2.5, image.cols * 8 / 10, image.rows / 2));19 20    // Gamma correction/gamma nonlinearity 伽马校正 -> https://www.cnblogs.com/sdu20112013/p/11597171.html21    cv::Mat look_up_table(1, 256, CV_8U);22    uchar *p = look_up_table.ptr();23    float gamma = 0.7;24    for (int i = 0; i < 256; ++i)25        p[i] = cv::saturate_cast<uchar>(pow(i / 255.0, gamma) * 255.0);26    cv::LUT(image, look_up_table, image);27 28    cv::Mat gray;29    // 转化为灰度图像30    cv::cvtColor(image, gray, CV_BGR2GRAY, 0);31    // 阈值操作32    cv::threshold(gray, gray, inRange_threshold, 255, CV_THRESH_BINARY);33    //膨胀及腐蚀34    cv::erode(gray, gray, cv::Mat(), cv::Point(-1, -1), 2);35    cv::dilate(gray, gray, cv::Mat(), cv::Point(-1, -1), 3);36    // 平滑图像37    cv ::medianBlur(gray, gray, 5);38    // 检测并画出关键点39    cv::drawKeypoints(image, dector(gray), image, cv::Scalar(0, 0, 255), cv::DrawMatchesFlags::DRAW_RICH_KEYPOINTS);40    out = image;41    return true;42}
